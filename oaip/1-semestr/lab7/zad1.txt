#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <math.h>

int main(){
    char a[80], b[80];
    int a1 = 0, b1 = 0;
    scanf("%s %s", a, b);
    for (int i = 0; i < strlen(a);i++){
        if (a[i]=='0') continue;
        a[i]= a[i] - '0';
        a1 = a1 + a[i]*pow(10,strlen(a)-i-1);
    }
    for (int i = 0; i < strlen(b);i++){\
        if (a[i]=='0') continue;
        b[i]= b[i] - '0';
        b1 = b1 + b[i]*pow(10,strlen(b)-i-1);
    }
    
    printf("%d",b1+a1);
}
